#include <iostream>
#include <algorithm>

using namespace std;

int main() {
	int arr[9][9];

	for (int i = 0; i < 9; i++) {
		for (int j = 0; j < 9; j++) {
			cin >> arr[i][j];
		}
	}
	
	int max = 0;
	int row = 0;
	int column = 0;

	for (int i = 0; i < 9; i++) {
		for (int j = 0; j < 9; j++) {
			if (max < arr[i][j]) {
				max = arr[i][j];
				row = i;
				column = j;
			}
		}
	}

	cout << max << "\n";
	cout << row+1<< " " << column+1;
	return 0;
}
