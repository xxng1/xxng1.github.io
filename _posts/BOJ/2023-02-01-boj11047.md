#include <iostream>

using namespace std;
int arr[10] = {0,};

int main(){
    int N, K;
    cin >> N >> K;

    for(int i=0; i<N; i++){
        cin >> arr[i];
    }
    int cnt = 0;

    while(K!=0){
        for(int i=N-1; i>=0; i--){
            if(K==0) break;
            int rem = K/arr[i];
            K %= arr[i];
            cnt += rem;
        }
    }
    cout << cnt;
}
